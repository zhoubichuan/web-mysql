# 三.mysql 命令

net start MySQL # 启动
net stop MySQL  # 停止

mysql -h 127.0.0.1 -P 3306 -uroot -p123456 # 连接mysql
exit

use text #切换数据库

show tables # 显示有哪些表
show tables form mysql

select database() # 显示当前数据库

DESC user # 查询表结构
